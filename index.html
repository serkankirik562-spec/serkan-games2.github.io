🚗 Serkan Online Racing

const express=require("express"); const http=require("http"); const {Server}=require("socket.io"); const app=express(); const server=http.createServer(app); const io=new Server(server); app.use(express.static("public")); let players={}; /* CONNECT */ io.on("connection",(socket)=>{ players[socket.id]={ x:300, y:600 }; /* MOVE */ socket.on("move",(data)=>{ if(players[socket.id]){ players[socket.id].x=data.x; players[socket.id].y=data.y; } }); /* DISCONNECT */ socket.on("disconnect",()=>{ delete players[socket.id]; }); }); /* BROADCAST */ setInterval(()=>{ io.emit("players",players); },50); server.listen(3000,()=>{ console.log("🚗 Racing Server çalışıyor"); });
